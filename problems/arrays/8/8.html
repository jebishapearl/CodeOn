Counting Prime Numbers in an Array

Problem Statement

Given an array of integers, count how many elements in the array are prime numbers.

Example 1

Input:
5
2 3 4 5 10

Output:
3

Explanation:
The prime numbers in the array are 2, 3, and 5. Hence, the output is 3.

Example 2

Input:
6
6 7 8 9 10 11

Output:
2

Explanation:
The prime numbers in the array are 7 and 11. Hence, the output is 2.

bool is_prime(int num) {
    if (num <= 1) return false;
    for (int i = 2; i * i <= num; i++) {
        if (num % i == 0) {
            return false;
        }
    }
    return true;
}

int main() {
    int n, prime_count = 0;
    scanf("%d", &n);
    
    int arr[n];
    for (int i = 0; i < n; i++) {
        scanf("%d", &arr[i]);
    }
    
    for (int i = 0; i < n; i++) {
        if (is_prime(arr[i])) {
            prime_count++;
        }
    }
    
    printf("%d\n", prime_count);
    return 0;
}
